# Flujo de Checkout y Ejemplo de Backend (Node.js/Express) ## 1. Diagrama de Flujo Lógico: Registro y Compra ```mermaid graph TD A[Usuario en Carrito] --> B{¿Hizo Click en Checkout?} B -- Sí --> C{¿Sesión Activa?} C -- No --> D[Redirigir a /registro] D --> E[Formulario: Nombre, Email, Teléfono, Tipo de Cliente] E --> F[Endpoint POST /api/register] F --> G[Guardar Usuario en DB (PostgreSQL)] G --> H[Redirigir a /carrito con Sesión] H --> I[Finalizar Pedido (POST /api/orders)] C -- Sí --> I ``` ## 2. Código de Ejemplo (Backend) ```javascript const express = require('express'); const router = express.Router(); const db = require('./db_connection'); // Instancia de Sequelize o Knex /** * Endpoint de Registro de Usuario * Captura: Nombre, Email, Teléfono, y Tipo de Cliente (Taller, Flota, Particular) */ router.post('/api/register', async (req, res) => { const { name, email, password, phone, customerType } = req.body; try { // 1. Guardar el nuevo usuario en la base de datos const newUser = await db('users').insert({ name, email, password_hash: await hash(password), // Función de hashing sugerida phone, customer_type: customerType, // ENUM: 'workshop', 'fleet', 'individual' created_at: new Date() }).returning('*'); // 2. Generar JWT o sesión (opcional dependiendo de la estrategia de auth) const token = generateToken(newUser[0]); // 3. Responder con el usuario y el token para persistencia en cliente res.status(201).json({ user: newUser[0], token }); } catch (error) { res.status(500).json({ message: 'Error al registrar el usuario', error }); } }); /** * Endpoint de Creación de Pedidos * Vincula el carrito con el ID del usuario */ router.post('/api/orders', async (req, res) => { const { userId, cartItems, totalAmount, warehouseId } = req.body; try { // Iniciar transacción para asegurar consistencia await db.transaction(async (trx) => { // 1. Crear el encabezado del pedido const [orderId] = await trx('orders').insert({ user_id: userId, total_amount: totalAmount, status: 'pending', warehouse_id: warehouseId, created_at: new Date() }).returning('id'); // 2. Guardar los detalles de los repuestos (Detalle del Pedido) const orderLines = cartItems.map(item => ({ order_id: orderId, product_id: item.productId, quantity: item.quantity, price_unit: item.price })); await trx('order_details').insert(orderLines); // 3. (Opcional) Descontar stock del almacén seleccionado for (const item of cartItems) { await trx('inventory') .where({ product_id: item.productId, warehouse_id: warehouseId }) .decrement('stock_quantity', item.quantity); } }); res.status(201).json({ message: 'Pedido creado exitosamente' }); } catch (error) { res.status(500).json({ message: 'Error al procesar el pedido', error }); } }); module.exports = router; ```
